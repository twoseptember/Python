def armnum(num):
    sum = 0
    temp = num
    while temp > 0:  
        digit = temp % 10  
        sum += digit**3  
        temp //= 10  

    if num == sum:
        print(num, "is an Armstrong Number")
    else:
        print(num, "is not an Armstrong Number")

def palnum(num):
    sum = 0
    temp = num
    while num != 0:  
        rem = num % 10  
        sum = rem + sum * 10  
        num //= 10  

    if temp == sum:
        print(temp, "is a Palindrome Number")
    else:
        print(temp, "is not a Palindrome Number")

num = int(input("Enter any Number: "))

armnum(num)  
palnum(num)
